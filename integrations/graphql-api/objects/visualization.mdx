<ParamField path="id" type="String!" required>

<ParamField path="pagination" type="Pagination!" required>

<ParamField path="totalRows" type="Int">

<ParamField path="columnDisplayFormats" type="[DisplayFormat!]!" required>

<ParamField path="columnOrder" type="[String!]!" required>

<ParamField path="dimensions" type="[String!]!" required>

<ParamField path="uniqueDimensions" type="[String!]!" required>

<ParamField path="measures" type="[String!]!" required>

<ParamField path="filters" type="[FilterOperation!]!" required>
